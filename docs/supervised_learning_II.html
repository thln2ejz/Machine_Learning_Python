def synth_data(n_samples, sigma):
    X = np.random.uniform(low=0, high=2, size=n_samples)
    X = np.sort(X)
    y = np.sin(4*X) + np.random.normal(loc=0, scale=sigma, size=n_samples) 
    return X, y
